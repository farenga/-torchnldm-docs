[docs]class NLDMBase(nn.Module, ABC):
    '''
    Neural Latent Dynamics Model abstract base class. It defines the basic structure of a NLDM, composed of an encoder, 
    a latent NDE and a decoder. It provides a common layout for NLDMs and template methods, based on abstract methods that
    need to be implemented by subclasses, thus ensure compatibility with the Trainer and Evaluator classes.
        
    Args:
        :encoder: (nn.Module), maps the state u(t) to its latent representation z(t).
        :decoder: (nn.Module), maps the latent state z(t) to back to high-dimensions, namely u(t).
        :latent_nde: (NeuralDE), evolves the latent state z(t), from t_0, to t_f.
        :integration_method: (std), numerical scheme used for the latent time-integration (those available in torchdiffeq (ODEs), torchsde (SDEs)).
        :adjoint: (bool), flag for enabling the adjoint method (https://arxiv.org/abs/1806.07366) 
        :device: (torch.device), device where the model will be loaded.            
    '''

    
    def __init__(self, 
                 encoder: nn.Module, 
                 decoder: nn.Module, 
                 latent_nde: NeuralODE or NeuralSDE,
                 nde_integration_method: str = 'euler',
                 nde_adjoint: bool = False,
                 device: torch.device = torch.device('cpu')):
        
        super(NLDMBase, self).__init__()

        self.encoder = encoder
        self.decoder = decoder

        if type(latent_nde) in (NeuralODE, NeuralSDE):
            self.latent_nde = latent_nde
            self.nde_type = latent_nde.nde_type
        else:
            raise TypeError(f'latent_nde type must be either NeuralODE or NeuralSDE, got instead {type(latent_nde)}.')
        
        
        self.__nde_integration_method = nde_integration_method
        self.__nde_adjoint = nde_adjoint 
        
        self.__latent_guidance = True
        self.__bidirectional = False

        self.device = device           
        self.to(self.device)
        self.eval()
    
    @property
    def nde_integration_method(self) -> str:
        '''
        Integration method to perform latent integration.
        Available methods are those implemented within TorchDiffEq (NeuralODE) or TorchSDE (NeuralSDE).
        '''
        return self.__nde_integration_method

    @property
    def nde_adjoint(self) -> bool:
        '''
        Flag enabling the adjoint sensitivity method.
        '''
        return self.__nde_adjoint
    
    @property
    def latent_guidance(self) -> bool:
        '''
        Flag enabling the latent guidance training scheme.
        '''
        return self.__latent_guidance
    
    @property
    def bidirectional(self) -> bool:
        '''
        Flag enabling the bidirectional training scheme.
        '''
        return self.__bidirectional

    @nde_integration_method.setter
    def nde_integration_method(self, nde_integration_method:str):
        if self.training:
            raise Exception('Integration method cannot be changed at training time.')  
        else: 
            self.__nde_integration_method = nde_integration_method

    def adjoint(self, mode: bool) -> None:

        else: 
            self.__nde_adjoint = mode

        if self.training:

        if self.training:

[docs]    def training_epoch(self, 
                       dataset: DataLoader, 
                       optimizer: torch.optim, 
                       batch_sizes: list[int], 
                       enable_amp: bool = False) -> float:        
        '''
        Template method defining the general training routine for one epoch. It is based on the abstract methods defining
        the specific NLDM behavior, which need to be implemented in the child classes, inheriting from the base class.

        Args:
            :dataset: (DataLoader), dataset object used for training purposes.
            :optimizer: (torch.optim), torch optimizer (e.g. torch.optim.Adam).
            :batch_sizes: (list[int]), list of batch sizes, more specifically [batch_size_params, batch_size_timesteps, batch_size_nodes].
            :enable_amp: (bool), flag for enabling Automatic Mixed Precision.

        Returns:
            :epoch_loss: (float), value of the loss.
        '''

        self.train()
        
        if enable_amp:
            # Scale gradients if Automatic Mixed Precision is enabled
            grad_scaler = torch.cuda.amp.GradScaler()

        section_id, sub_trajectories_t0_indexes, n_subtrajectories = dataset.get_train_t0_indices(timesteps=batch_sizes[1], shuffle=True)
        epoch_loss = 0.

        for t0_index in sub_trajectories_t0_indexes:
            
            try:
                optimizer.zero_grad()
                
                if enable_amp:
                    with torch.cuda.amp.autocast():
                        loss = self.get_batch_loss(dataset, section_id, t0_index, batch_sizes)
                        grad_scaler.scale(loss).backward()
                        grad_scaler.step(optimizer)
                        grad_scaler.update()
                else:
                    loss = self.get_batch_loss(dataset, section_id, t0_index, batch_sizes)
                    loss.backward()
                    optimizer.step()

                epoch_loss += loss.item()/n_subtrajectories

            except RuntimeError as e:
                if 'out of memory' in str(e):
                    print('WARNING: ran out of memory, skipping batch')
                    for p in self.parameters():
                        if p.grad is not None:
                            del p.grad
                    torch.cuda.empty_cache()
                    continue
        
        return epoch_loss

    def validation_epoch(self, 
                         dataset: DataLoader, 
                         batch_sizes: list[int],
                         enable_amp: bool = False) -> float:        
        '''
        Template method defining the general validation routine for one epoch. It is based on the abstract methods defining
        the specific NLDM behavior, which need to be implemented in the child classes, inheriting from the base class.

        Args:
            :dataset: (DataLoader), dataset object used for training purposes.
            :batch_sizes: (list[int]), list of batch sizes, more specifically [batch_size_params, batch_size_timesteps, batch_size_nodes].
            :enable_amp: (bool), flag for enabling Automatic Mixed Precision.

        Returns:
            :epoch_loss: (float), value of the loss.
        '''
        assert 'validation' in dataset.sections, 'Validation cannot be performed: provided dataset is not splitted, time_validation_split must be specified when loading the dataset.'

        self.eval()
        with torch.cuda.amp.autocast(enabled=enable_amp):
            epoch_loss = 0.
            n_subtrajectories =  dataset.Nt - dataset.Nt_train - batch_sizes[1]

            for t0_index in range(n_subtrajectories):
                loss = self.get_batch_loss(dataset, 'validation', t0_index, batch_sizes)
                epoch_loss += loss.item()/n_subtrajectories
            
        return epoch_loss

    def evaluate(self, 
                 dataset: DataLoader, 
                 t0_index: int,
                 param_index: int,
                 timesteps: int,
                 enable_scaling: bool = False) -> list[float]:     
        '''
        Template method defining the general evaluation routine for a single trajectory prediction. 
        It is based on the predict_trajectory abstract method.

        Args:
            :dataset: (DataLoader), dataset object used for training purposes.
            :t0_index: (int), index of the initial value (<Nt).
            :param_index: (int), index of the parameter instance (<N).
            :timesteps: (int), length of the prediction window (<N).
            :enable_scaling: (bool), flag to enable pre- and post-processing via the model's internal scaler.

        Returns:
            :epoch_loss: (float), value of the loss.
        '''
        self.eval()
        ut, ut_pred, pred_t = self.predict_trajectory(dataset, t0_index, param_index, timesteps, enable_scaling)
        pred_error = error(ut, ut_pred).item()
        pred_rel_error = relative_error(ut, ut_pred).item()

        return pred_error, pred_rel_error, pred_t

[docs]    def POD_layer(self, u: Tensor, position: str = 'input') -> Tensor:
        '''
        Method for handling the POD layer, to perform forward and inverse projection, depending
        on the provided 'position' argument.

        Args:
            :u: (Tensor): input to be projected
            :position: (str): if 'input' it applies U, while if 'output' U^T (transposed) is applied.
        '''
        if hasattr(self, 'U'):
            if position == 'input':
                return u@self.U
            elif position == 'output':
                return u@self.U.T
        else:
            return u

[docs]    def forward(self, 
                u0: Tensor, 
                mut0:Tensor, 
                t: Tensor, 
                return_latent: bool = False, 
                preprocess: bool = False, 
                postprocess: bool = False) -> Tensor or tuple[Tensor]:
        '''
        Forward method defining the inference behavior of the NLDM.

        Args: 
            :u0: (Tensor): discretized state initial value, of shape = (B,Nh).
            :mut0: (Tensor): parameters tensor, concatenated with the initial timestep (as provided by PDEGen), of shape = (B,Np).
            :t: (Tensor): 1d tensor of timesteps to perform latent integration, of shape (T,).
            :return_latent: (bool): flag to return the latent dynamics z(t), evalued at the timesteps t.
            :preprocess: (bool): flag to perform input features scaling.
            :postprocess: (bool): flag to perform output scaling.

        Returns:
            :ut: (Tensor): evolution of the high-dimensional state, of shape shape = (T,B,Nh).
            :zt: (Tensor): evolution of the latent state,  of shape shape = (T,B,latent_dim).

        Shapes:
            - B: batch size
            - Nh: degrees of freedom of the discretized state
            - T: number of timesteps involved in the prediction (length of the prediction window)
            - latent_dim: dimension of the latent state vector
        '''
        if preprocess:
            u0, mut0, t = self.preprocess(u0, mut0, t)
        
        u0 = self.POD_layer(u0, 'input')
        z0 = self.encoder(u0, mut0)
        zt = self.latent_nde(z0, t, self.nde_integration_method)
        ut = self.decoder(zt)
        ut = self.POD_layer(ut, 'output')

        if postprocess:
            ut = self.postprocess(ut)

        return (ut, zt) if return_latent else ut

[docs]    def loss(self, 
             u0: Tensor, 
             mut0: Tensor, 
             t: Tensor, 
             ut: Tensor, 
             mut: Tensor) -> Tensor:
        
        u0 = self.POD_layer(u0, 'input')
        ut_ = self.POD_layer(ut, 'input')

        if self.latent_guidance:
            zt = self.encoder(ut_, mut)
            ut_pred = self.decoder(zt)
            ut_pred = self.POD_layer(ut_pred, 'output')
            loss = self.loss_criterion(ut, ut_pred) # u0,...,uT autoencoding loss term
            zt_pred = self.latent_nde(zt[0,...], t, self.nde_integration_method) 
            loss += self.loss_criterion(zt, zt_pred) # latent guiding loss term

            if self.bidirectional:
                t_backward = torch.flip(t,[0])
                zt_backward = torch.flip(zt,[0])
                zt_backward_pred = self.latent_nde(zt[-1,...], t_backward, self.nde_integration_method) 
                loss += self.loss_criterion(zt_backward, zt_backward_pred) # latent backward dynamics loss term
        else:
            ut_pred = self.forward(u0, mut0, t)
            loss = self.loss_criterion(ut, ut_pred) # observable dynamics loss term
            loss += self.loss_criterion(self.decoder(self.encoder(u0, mut0)),u0) # u0 autoencoding loss term
            
            if self.bidirectional:
                t_backward = torch.flip(t,[0])
                ut_backward = torch.flip(ut_pred,[0])
                ut_backward_pred = self.forward(ut_backward[0,...], mut[-1,...], t_backward)
                loss += self.loss_criterion(ut_backward, ut_backward_pred) # observable backward dynamics loss term

        # ** To be assessed from a theoretical point of view **
        # Stability term
        # implemented as {f(z*)==0}*max(λ(Jf(z*)),0)
        #z0 = zt[0,...]
        #loss += (torch.isclose(self.latent_nde.f.f(z0),torch.zeros_like(z0)).detach()*torch.max(eigvals(vmap(jacrev(self.latent_nde.f.f))(z0)).real, torch.zeros_like(z0))).sum()

        return loss

class NLDFConfig(ConfigBase):
    '''
    Neural Latent Dynamics Field configuration class.
        
    Attributes:
        :n: (int): dimension u(x,t;mu) codomain.
        :d: (int): number of dimension of spatial domain (x).
        :Np: (int): total number of parameters + time (Nmu + 1)
        :latent_dim: (int): dimension of the latent state z(t).
        :nde_type: (str): type of latent neural differential equation (ode/sde)
        :integration_method: (std): numerical scheme used for the latent NDE time-integration.
        :adjoint: (bool): flag for enabling the adjoint method (https://arxiv.org/abs/1806.07366) 
        :device: (torch.device): cpu/gpu(cuda)
    '''
    # Dimensions
    n: int
    d: int
    Np: int
    latent_dim: int
    # Autoencoder
    ae_activation: str = 'elu'
    encoder_depth: int = 3
    encoder_hidden_dim: int = 256
    encoder_architecture: str = 'hadamard'
    decoder_depth: int = 3
    decoder_hidden_dim: int = 256

[docs]    def forward(self, 
                u0: Tensor, 
                mut0: Tensor,
                x: Tensor, 
                t: Tensor, 
                return_latent: bool = False, 
                preprocess: bool = False, 
                postprocess: bool = False) -> Tensor or tuple[Tensor]:
        '''
        Forward method defining the inference behavior of the NLDF.

        Args: 
            :u0: (Tensor): discretized state initial value, of shape = (B,M,n).
            :mut0: (Tensor): parameters tensor, concatenated with the initial timestep (as provided by PDEGen), of shape = (B,M,Np).
            :x: (Tensor): nodes coordinates tensor of shape = (B,M,d).
            :t: (Tensor): 1d tensor of timesteps to perform latent integration, of shape (T,).
            :return_latent: (bool): flag to return the latent dynamics z(t), evalued at the timesteps t.
            :preprocess: (bool): flag to perform input features scaling.
            :postprocess: (bool): flag to perform output scaling.

        Returns:
            :ut: (Tensor): evolution of the high-dimensional state, of shape shape = (T,B,Nh).
            :zt: (Tensor): evolution of the latent state,  of shape shape = (T,B,latent_dim).

        Shapes:
            - B: batch size (params)
            - M: batch size (nodes coordinates)
            - n: dim of u's codomain
            - d: dim of u's spatial-domain
            - T: number of timesteps involved in the prediction (length of the prediction window)
            - latent_dim: dimension of the latent state vector
        '''
        if preprocess:
            u0, mut0, x, t = self.preprocess(u0, mut0, x, t)

        if x.dim()==3:
            #[B,M,d] -> [T,B,M,d]
            xt = x.expand(t.shape[0],-1,-1,-1)
        elif x.dim()==2:
            #[M,d] -> [T,M,d]
            xt = x.expand(t.shape[0],-1,-1)
        
        z0 = self.encoder(u0, mut0, x)
        zt = self.latent_nde(z0, t, self.nde_integration_method)
        ut = self.decoder(zt, xt)

        if postprocess:
            ut = self.postprocess(ut)

        return (ut, zt) if return_latent else ut

[docs]    def loss(self, 
             u0: Tensor, 
             mut0: Tensor,
             x: Tensor, 
             t: Tensor, 
             ut: Tensor, 
             mut: Tensor) -> Tensor:
        
        if self.latent_guidance:
            xt = x.expand(t.shape[0],-1,-1,-1)
            zt = self.encoder(ut, mut, xt)
            ut_pred = self.decoder(zt, xt)
            loss = self.loss_criterion(ut, ut_pred) # u0,...,uT autoencoding loss term
            zt_pred = self.latent_nde(zt[0,...], t, self.nde_integration_method) 
            loss += self.loss_criterion(zt, zt_pred) # latent guiding loss term

            if self.bidirectional:
                t_backward = torch.flip(t,[0])
                zt_backward = torch.flip(zt,[0])
                zt_backward_pred = self.latent_nde(zt[-1,...], t_backward, self.nde_integration_method) 
                loss += self.loss_criterion(zt_backward, zt_backward_pred) # latent backward dynamics loss term
        else:
            ut_pred = self.forward(u0, mut0, x, t)
            loss = self.loss_criterion(ut, ut_pred)                                 # observable dynamics loss term
            loss += self.loss_criterion(self.decoder(self.encoder(u0,mut0,x),x),u0) # u0 autoencoding loss term

            if self.bidirectional:
                t_backward = torch.flip(t,[0])
                ut_backward = torch.flip(ut_pred,[0])
                ut_backward_pred = self.forward(ut_backward[0,...], x, mut[-1,...], t_backward)
                loss += self.loss_criterion(ut_backward, ut_backward_pred)  # observable backward dynamics loss term
        return loss
