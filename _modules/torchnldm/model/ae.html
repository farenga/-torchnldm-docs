[docs]class StateEncoder(nn.Module):
    '''
    State encoder for producing an embedding of the spatially-discretized state uh(t).
    Consists of a feed-forward convolutional architecture, possibly equipped with residual connections.
    For the parameterized version consider using the PStateEncoder class.

    Args:
        :Nh: (int), number of degrees of freedom of the spatially discretized state uh(t).
        :Np: (int), number of parameters (Nmu) + 1.
        :latent_dim: (int), dimension of the latent state z(t).
        :conv_dim: (int), dimensionality of the emlpoyed convolutions (1d,2d).
        :conv_depth: (int), number of convolutional layers (or blocks) in the state decoder.
        :in_channels: (int), number of channels of the input state.
        :hidden_channels: (int), number of output channels produced by the first convolutional layer (or block).
        :reshape_dim: (int), to be specified when applying Conv2d to 1d vectors of size `Nh`, so reshape_dim must be `int(sqrt(Nh))`.
        :residual_blocks: (bool), if True blocks with two convolutional layers equipped with skip connections are used, instead of single convolutional layers.
        :mlp_state_hidden_dim: (int), dimensionality of the state-embedding MLP hidden layers.
        :param_emb: (str), method to combine state and parameters embeddings ('multiply', 'project', 'sum').
        :mlp_param_depth: (int), number of hidden layers of the parameters-embedding MLP.
        :mlp_param_hidden_dim: (int), dimensionality of the parameters-embedding MLP hidden layers.
        :mlp_time_depth: (int), number of hidden layers of the time-embedding MLP.
        :mlp_time_hidden_dim: (int), dimensionality of the time-embedding MLP hidden layers.
        :activation: (str), activation of the state and parameter encoders.
    
    Architecture:

    '''

    def __init__(self,
                 Nh: int,
                 Np: int,
                 latent_dim: int,
                 conv_dim: int,
                 conv_depth: int,
                 in_channels: int,
                 hidden_channels: int,
                 reshape_dim: int or tuple = None,
                 residual_blocks: bool = False,
                 mlp_state_hidden_dim: int = 256,
                 param_emb: str = 'multiply',
                 mlp_param_depth: int = 2,
                 mlp_param_hidden_dim: int = 256,
                 mlp_time_depth: int = 2,
                 mlp_time_hidden_dim: int = 256,
                 activation: str = 'elu') -> None:
        
        super().__init__()

        self.Nh = Nh
        self.Np = Np
        self.latent_dim = latent_dim
        self.conv_dim = conv_dim
        self.conv_depth = conv_depth
        self.in_channels = in_channels
        self.hidden_channels = hidden_channels
        self.reshape_dim = reshape_dim
        self.residual_blocks = residual_blocks
        self.mlp_state_hidden_dim = mlp_state_hidden_dim
        self.param_emb = param_emb
        self.mlp_param_depth = mlp_param_depth
        self.mlp_param_hidden_dim = mlp_param_hidden_dim
        self.mlp_time_depth = mlp_time_depth
        self.mlp_time_hidden_dim = mlp_time_hidden_dim
        self.activation = activation

        # Assert convolutions type
        assert conv_dim in (1,2), f'conv_dim must be either 1 or 2, got instead {conv_dim}.'
        
        if conv_dim==2 and reshape_dim is None:
            self.reshape_dim = int(math.sqrt(Nh))
            warnings.warn('Reshape dimension not specified, using reshape_dim = sqrt(Nh).')
        
        # Embedding Method 
        # multiply: element-wise product between the state and parameters embeddings
        # sum: element-wise sum between the state and parameters embeddings
        # project: computes the projection of the state embedding along the parameters embedding vector
        self.__param_emb_methods = {'multiply': lambda a,b : a*b,
                                    'sum': lambda a,b : a+b,
                                    'project': lambda a,b : self.__inner(a,b)*b/self.__inner(b,b)}
        assert param_emb in self.__param_emb_methods.keys(), f'Available parameters embedding methods are {self.__param_emb_methods.keys()}, got instead {param_emb}.'
        self.__emb_method = self.__param_emb_methods[param_emb]

        # Construction of the convolutional net to produce intermediate state embedding
        conv_layer = ConvResBlock if self.residual_blocks else ConvLayer
        self.input_block = Downsample(conv_layer(in_channels, hidden_channels, conv_dim, activation = activation), conv_dim)

        middle_layers = []
        out_ch = hidden_channels
        assert conv_depth>=1, 'At least 1 hidden layer is required, consisting of a network with 1 input and 1 output layers.'
        for i in range(conv_depth-1):
            in_ch = hidden_channels*(2**i)
            out_ch = hidden_channels*(2**(i+1))
            middle_layers.append(Downsample(conv_layer(in_ch, out_ch, conv_dim, activation = activation), conv_dim))
        self.middle_block = nn.Sequential(*middle_layers)
        self.output_block = ConvLayer(out_ch, out_ch, conv_dim, last = True)

        # MLPs instantiation
        mlp_in_dim = self.__forward_conv(torch.rand(Nh)).shape[-1]
        self.mlp_state = MLP(mlp_in_dim, latent_dim, mlp_state_hidden_dim, 1, activation)
        self.mlp_param = MLP(Np-1, latent_dim, mlp_param_hidden_dim, mlp_param_depth, activation)
        self.mlp_time = MLP(1, latent_dim, mlp_time_hidden_dim, mlp_time_depth, activation)
        
    def __in_reshape(self, u: Tensor) -> Tensor:
        '''
        Performs input reshaping, depending on the type of convolutions employed.
        '''
        if self.conv_dim==2:
            if isinstance(self.reshape_dim, int):
                return u.view(-1, self.in_channels, self.reshape_dim, self.reshape_dim)
            elif isinstance(self.reshape_dim, tuple):
                return u.view(-1, self.in_channels, self.reshape_dim[1], self.reshape_dim[0])
        elif self.conv_dim==1 and self.reshape_dim is not None:
            return u.view(-1, self.in_channels, self.reshape_dim)
        elif self.conv_dim==1:
            return u.view(-1, 1, self.Nh)

    def __out_reshape(self, h: Tensor, u: Tensor) -> Tensor:
        '''
        Performs state embedding reshape method, output of the convolutional trunk.
        Depends on the input state shape.
        '''
        if u.dim()==3:
            # Input u of shape [T,B,Nh]
            T, B, _= u.shape
            out_shape = [T,B,-1]
        elif u.dim()==2:
            # Input u of shape [B,Nh]
            B, _ = u.shape
            out_shape = [B,-1]
        elif u.dim()==1:
            # Input u of shape [Nh]
            out_shape = [1,-1]
        
        return h.view(out_shape)
    
    def __inner(self, a: Tensor, b: Tensor) -> Tensor:
        '''
        Computing inner product between tensors.
        Depending on the batching (T,B) and architecture (dense, fully conv), we
        have to deal with differently shaped tensor, which are handled by this method,
        to compute the inner (dot) product.
        '''
        assert a.dim()==b.dim(), f"Dimensions must coincide when computing tensor inner product, got a ({a.dim()}) and b ({b.dim()})."
        if a.dim() == 2:
            #[B,latent_dim]
            inner = torch.einsum('ij,ij->i',a,b)
        elif a.dim() == 3:
            #[T,B,latent_dim]
            inner = torch.einsum('ijk,ijk->ij',a,b)
        return multiple_unsqueeze(inner, a.dim()-inner.dim()) 
    
    def __forward_conv(self, u: Tensor) -> Tensor:
        '''
        Defines the forward data flow for the convolutional trunk, to produce the the state embedding.
        '''
        h = self.__in_reshape(u)    
        h = self.input_block(h)
        h = self.middle_block(h)
        h = self.output_block(h)
        return self.__out_reshape(h, u)
        

[docs]    def forward(self, u: Tensor, mut: Tensor) -> Tensor:
        '''
        Forward method defining the inference scheme of the state encoder module.
        '''
        if mut.dim()==1:
            mut = mut.unsqueeze(0)
        # Compute state embedding
        u_emb = self.__forward_conv(u)        
        u_emb = self.mlp_state(u_emb)
        # Compute parameters embedding
        mu_emb = self.mlp_param(mut[...,:-1])
        # Compute time embedding
        t0_emb = self.mlp_time(mut[...,-1:])
        return self.__emb_method(u_emb, mu_emb) + t0_emb # Combine embeddings

    '''
    Field Encoder class.

    Args:
        :n: (int), dimension of the continuos state u(x,t;mu).
        :Np: (int), dimension of the parameters vector (mu,t), dim=(Nmu + 1).
        :d: (int), dimension of the spatial domain (n coordinates).
        :latent_dim: (int), dimension of the output latent encoding.
        :depth: (int), number of layers of the MLPs.
        :hidden_dim: (int), MLPs' hidden layers size.
        :activation: (str), activation of the state and parameter encoders.
        :architecture: (str), defines the network architecture
    '''
    def __init__(self,
                 n: int,
                 Np: int,
                 d: int, 
                 latent_dim: int,
                 depth: int = 3,
                 hidden_dim: int = 64,
                 activation: str = 'elu',
                 architecture: str = 'hidden_sum') -> None:
        
        super().__init__()
        self.n = n
        self.Np = Np
        self.d = d
        self.latent_dim = latent_dim
        self.depth = depth
        self.hidden_dim = hidden_dim
        self.activation = activation
        self.architecture = architecture

        if self.architecture in _hidden_op_layers.keys():
            self.net = HiddenOpNet(input_dim = n+Np,
                                   cond_input_dim = d,
                                   output_dim = latent_dim,
                                   hidden_dim = hidden_dim, 
                                   depth = depth,
                                   activation = activation,
                                   positional_encoding = True,
                                   architecture = architecture)
        elif self.architecture == 'mlp':
            self.net = MLP(n+Np+d, latent_dim, hidden_dim, depth, activation)
        else:
            raise ValueError(f'Available networks types are: {_hidden_op_layers.keys()} and "mlp", got instead {architecture}.')

[docs]class StateDecoder(nn.Module):
    '''
    State decoder for decoding a sequence of latent encodings z(t0;mu),...,z(tf;mu) into the high dimensional
    spatially-discretized states uh(t0;mu),...,uh(tf;mu).
    Consists of an input MLP, producing an intermediate representation, which is then reshaped and processed 
    by a feed-forward convolutional architecture, possibly equipped with residual connections.
    
    Args:
        :Nh: (int), dimension of the output spatially discretized state uh(t;mu).
        :latent_dim: (int), dimension of the input latent encodings z(t;mu).
        :conv_dim: (int), dimensionality of the emlpoyed convolutions (1d,2d).
        :conv_depth: (int), number of convolutional layers (or blocks) in the state decoder.
        :out_channels: (int), number of output channels of the decoded state.
        :hidden_channels: (int), number of channels produced by the first convolutional layer (or block).
        :reshape_dim: (int), intermediate reshape dimension to pass from the MLP representation to the convolutional one.
        :residual_blocks: (bool), if True blocks with two convolutional layers equipped with skip connections are used, instead of single convolutional layers.
        :mlp_hidden_dim: (int), dimensionality of the MLP hidden layers.
        :activation: (str), activation of the input MLP and the convolutional network.
    '''

    def __init__(self,
                 Nh: int,
                 latent_dim: int,
                 conv_dim: int,
                 conv_depth: int,
                 out_channels: int,
                 hidden_channels: int,
                 reshape_dim: int or tuple = None,
                 residual_blocks: bool = False,
                 mlp_hidden_dim: int = None,
                 activation: str = 'elu') -> None:
        
        super().__init__()
        
        # Assert convolutions type
        assert conv_dim in (1,2), f'conv_dim must be either 1 or 2, got instead {conv_dim}.'
        
        if reshape_dim is None:
            if conv_dim==2:
                reshape_dim = int(math.sqrt(latent_dim))
                warnings.warn('Reshape dimension not specified, using reshape_dim = sqrt(latent_dim).')
            else:
                reshape_dim = latent_dim

        self.Nh = Nh
        self.latent_dim = latent_dim
        self.conv_dim = conv_dim
        self.out_channels = out_channels
        self.hidden_channels = hidden_channels
        self.conv_depth = conv_depth
        self.reshape_dim = reshape_dim
        self.mlp_hidden_dim = mlp_hidden_dim
        self.activation = activation

        # Output shape check
        if isinstance(self.reshape_dim,int):
            out_shape = out_channels*(reshape_dim*2**conv_depth)**conv_dim
        elif isinstance(self.reshape_dim, tuple):
            out_shape = out_channels*(reshape_dim[0]*2**conv_depth)*(reshape_dim[1]*2**conv_depth)
        assert  out_shape == Nh, f'StateDecoder: Output dimensions ({out_shape}) and Nh ({Nh}) do not match, given reshape_dim={reshape_dim} and depth={conv_depth}.'
        
        if mlp_hidden_dim is not None:
            if conv_dim==1:
                #self.mlp = MLP([latent_dim, hidden_channels*reshape_dim//2, hidden_channels*reshape_dim], activation)
                self.mlp = MLP([latent_dim, mlp_hidden_dim, reshape_dim], activation)
            elif conv_dim==2:
                #self.mlp = MLP([latent_dim, hidden_channels*reshape_dim**2//2, hidden_channels*reshape_dim**2], activation)
                if isinstance(reshape_dim,int):
                    self.mlp = MLP([latent_dim, mlp_hidden_dim, reshape_dim**2], activation)
                elif isinstance(reshape_dim, tuple):
                    self.mlp = MLP([latent_dim, mlp_hidden_dim, reshape_dim[0]*reshape_dim[1]], activation)
        else:
            self.mlp = nn.Identity()
            
        # Construction of the convolutional trunk
        conv_layer = ConvResBlock if residual_blocks else ConvLayer
        self.input_block = Upsample(conv_layer(hidden_channels, hidden_channels, conv_dim, activation=activation))

        middle_layers = []
        out_ch = hidden_channels
        assert conv_depth>=1, 'At least 1 hidden layer is required, consisting of a network with 1 input and 1 output layers.'
        for i in range(conv_depth-1):
            in_ch = hidden_channels//2**i
            out_ch = hidden_channels//2**(i+1)
            middle_layers.append(Upsample(conv_layer(in_ch, out_ch, conv_dim, activation=activation)))    
        self.middle_block = nn.Sequential(*middle_layers)
        
        self.output_block = ConvLayer(out_ch, self.out_channels, conv_dim, last=True)
    
    def __add_channels(self, h: Tensor) -> Tensor:
        '''
        Adds channels to the hidden state, previously processed by the input MLP.
        '''
        if h.dim()==1:
            return h.unsqueeze(-2).expand(self.hidden_channels,-1)
        elif h.dim()==2:
            return h.unsqueeze(-2).expand(-1,self.hidden_channels,-1)
        elif h.dim()==3:
            return h.unsqueeze(-2).expand(-1,-1,self.hidden_channels,-1)

    def __in_reshape(self, h: Tensor) -> Tensor:
        '''
        Reshapes the hidden state to be processed by the convolutional trunk.
        '''
        if self.conv_dim == 1:
            return h.view(-1, self.hidden_channels, self.reshape_dim)
        elif self.conv_dim == 2:
            if isinstance(self.reshape_dim, int):
                return h.view(-1, self.hidden_channels, self.reshape_dim, self.reshape_dim)
            elif isinstance(self.reshape_dim, tuple):
                return h.view(-1, self.hidden_channels, self.reshape_dim[1], self.reshape_dim[0])
            
    def __out_reshape(self, h: Tensor, z: Tensor) -> Tensor:
        '''
        Reshapes the output to comply with original dimensions (Nh).
        Depends on batching, time sequencing and convolutions' dimensions.
        '''
        if z.dim()==3:
            T, B, _ = z.shape
            return h.view(T, B, self.Nh).squeeze()
        elif z.dim()==2:
            B, _ = z.shape
            return h.view(B, self.Nh).squeeze()
        elif z.dim()==1:
            return h.view(self.Nh).squeeze()


[docs]    def forward(self, z: Tensor) -> Tensor:
        h = self.mlp(z)
        h = self.__add_channels(h)
        h = self.__in_reshape(h)
        h = self.input_block(h)
        h = self.middle_block(h)
        h = self.output_block(h)
        return self.__out_reshape(h,z)

    '''
    Field Decoder class.

    Args:
        :n: (int), dimension of the continuos state u(x,t;mu).
        :Np: (int), dimension of the parameters vector (mu,t), dim=(Nmu + 1).
        :d: (int), dimension of the spatial domain (n coordinates).
        :latent_dim: (int), dimension of the output latent encoding.
        :depth: (int), number of layers of the MLPs.
        :hidden_dim: (int), MLPs' hidden layers size.
        :activation: (str), activation of the state and parameter encoders.
    '''
    def __init__(self,
                 n: int,
                 d: int, 
                 latent_dim: int,
                 depth: int = 3,
                 hidden_dim: int = 64,
                 activation: str = 'elu',
                 architecture: str = 'hidden_sum') -> None:
        
        super().__init__()
        self.n = n
        self.d = d
        self.latent_dim = latent_dim
        self.depth = depth
        self.hidden_dim = hidden_dim
        self.activation = activation
        self.architecture = architecture

        if self.architecture in _hidden_op_layers.keys():
            self.net = HiddenOpNet(input_dim = latent_dim,
                                   cond_input_dim = d,
                                   output_dim = n,
                                   hidden_dim = hidden_dim, 
                                   depth = depth,
                                   activation = activation,
                                   positional_encoding = True,
                                   architecture = architecture)
        elif self.architecture == 'mlp':
            self.net = MLP(latent_dim+d, n, hidden_dim, depth, activation)
        elif self.architecture == 'mlp_no_coords':
            self.net = MLP(latent_dim, n, hidden_dim, depth, activation)
        else:
            raise ValueError(f'Available networks types are: {_hidden_op_layers.keys()} or "mlp" or "mlp_no_coords", got instead {architecture}.')
