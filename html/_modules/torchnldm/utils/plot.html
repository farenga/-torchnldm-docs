[docs]def get_gridpoints_2d(coords, n:int):
    
    x, y = coords[:,0].numpy(), coords[:,1].numpy()
    
    x_min, x_max = x.min(), x.max()
    u_min, u_max = y.min(), y.max()
    
    xi = np.linspace(x_min, x_max, n)
    yi = np.linspace(u_min, u_max, n)
    
    grid_x, grid_y = np.meshgrid(xi, yi)

    return grid_x, grid_y, x_min, x_max, u_min, u_max





[docs]def plot1(u: Tensor, 
          coords: Tensor, 
          title: str ='Solution', 
          figsize: tuple = (7,7), 
          filename: str = None, 
          res: int = 100, 
          cmap = cm.plasma):

    with torch.no_grad():

        grid_x, grid_y, x_min, x_max, u_min, u_max = get_gridpoints_2d(coords.detach().cpu(), res)
        
        grid_u = griddata(coords.detach().cpu(), u.detach().cpu(), (grid_x, grid_y), method='cubic')

        fig, axs = plt.subplots(1,1,figsize=figsize)
        
        plt0 = axs.imshow(grid_u, extent=(x_min, x_max, u_min, u_max), origin='lower', cmap=cmap)
        axs.set_title(title)
        axs.set_xlim(x_min, x_max)
        axs.set_ylim(u_min, u_max)
        
        fig.colorbar(plt0, ax=axs, shrink=0.7)
        
        if filename is not None:
            plt.savefig(filename, dpi=300, bbox_inches='tight', transparent=False, facecolor='white')

[docs]def plot_evolution(ut_true: Tensor, 
                   ut_pred: Tensor, 
                   t: Tensor, 
                   coords: Tensor, 
                   title: tuple = ('FOM','NLDM','Relative Error'),
                   figsize: tuple = (20,5), 
                   filename: str = None, 
                   res: int = 100, 
                   cmap=cm.plasma):

    # ut_true, ut_pred are of shape [N_timesteps, Nh]

    assert ut_true.shape[0]==ut_pred.shape[0], 'input tensors mush have same dimensions'
    n_timesteps = ut_true.shape[0]
    
    with torch.no_grad():

        if coords.shape[-1]==1:

            x = coords[:,0].detach().cpu()
            y = t.cpu()
            X, Y = torch.meshgrid(x, y,  indexing='ij')
            Z_true = ut_true.detach().cpu().t()
            Z_pred = ut_pred.detach().cpu().t()
            Z_err = relative_error_vectorized(ut_true, ut_pred).T.detach().cpu()
        
            fig = plt.figure(figsize=figsize)
            ax1 = fig.add_subplot(131)
            plt1 = ax1.contourf(X, Y, Z_true,res,cmap=cmap)
            ax1.set_title(title[0])
            ax1.set_xlabel('x')
            ax1.set_ylabel('t')

            ax2 = fig.add_subplot(132)
            plt2 = ax2.contourf(X, Y, Z_pred,res,cmap=cmap)

            ax2.set_title(title[1])
            ax2.set_xlabel('x')
            ax2.set_ylabel('t')

            ax3 = fig.add_subplot(133)
            plt3 = ax3.contourf(X, Y, Z_err,res,cmap=cmap)
            ax3.set_title(title[2])
            ax3.set_xlabel('x')
            ax3.set_ylabel('t')

            fig.colorbar(plt1, ax=ax1)
            fig.colorbar(plt2, ax=ax2)
            cbar = fig.colorbar(plt3, ax=ax3)
            cbar.formatter.set_powerlimits((0, 0))      

        elif coords.shape[-1]==2:

            grid_x, grid_y, x_min, x_max, u_min, u_max = get_gridpoints_2d(coords.detach().cpu(),res)
                        
            fig, axs = plt.subplots(2,n_timesteps,figsize=figsize)
            
            axs[0,0].set_ylabel(title[0])
            axs[1,0].set_ylabel(title[1])

            for i in range(n_timesteps):
                
                grid_true = griddata(coords.detach().cpu(), ut_true[i,...].detach().cpu(), (grid_x, grid_y), method='cubic')
                grid_pred = griddata(coords.detach().cpu(), ut_pred[i,...].detach().cpu(), (grid_x, grid_y), method='cubic')

                axs[0,i].imshow(grid_true, extent=(x_min, x_max, u_min, u_max), origin='lower', cmap=cmap)
                axs[0,i].set_title(f't={t[i]:,.4f}')
                axs[0,i].set_xticks([])
                axs[0,i].set_yticks([])

                axs[1,i].imshow(grid_pred, extent=(x_min, x_max, u_min, u_max), origin='lower', cmap=cmap)
                axs[1,i].set_xlim(x_min, x_max)
                axs[1,i].set_ylim(u_min, u_max)
                axs[1,i].set_xticks([])
                axs[1,i].set_yticks([])

        if filename is not None:
            plt.savefig(filename, dpi=300, bbox_inches='tight', transparent=False, facecolor='white')




[docs]def plot_evolution_error(ut_true: Tensor, 
                         ut_pred: Tensor, 
                         t: Tensor, 
                         title: str = 'Relative Error Evolution',
                         figsize: tuple = (7,5),
                         scilimit: int = -4, 
                         filename: str = None):
    
    t_err = relative_error_vectorized(ut_true, ut_pred).detach().cpu()
    t = t.detach().cpu().numpy()
    t_err_mean = t_err.mean(dim=-1).numpy()
    t_err_min = t_err.min(dim=-1).values.numpy()
    t_err_q1 = torch.quantile(t_err,q=.25,dim=-1).numpy()
    t_err_q3 = torch.quantile(t_err,q=.75,dim=-1).numpy()
    t_err_median = t_err.median(dim=-1).values.numpy()

    plt.figure(figsize=figsize)
    plt.plot(t,t_err_mean,'--k', label='Mean error')
    plt.plot(t,t_err_min, '-g', label='Min error')
    plt.plot(t,t_err_median, '-r', label='Median error')
    plt.plot(t,t_err_q1, '-m', label='Q1')
    plt.plot(t,t_err_q3, '-b', label='Q3')
    plt.xlabel(r'$t$')
    plt.ylabel(r'$\epsilon_{rel}$')
    plt.title(title)
    plt.ticklabel_format(axis='y', style='sci', scilimits=(scilimit,scilimit))
    plt.legend()

    if filename is not None:
        plt.savefig(filename, dpi=300, bbox_inches='tight', transparent=False, facecolor='white')
