class TrainerConfig(ConfigBase):
    '''
    Configuration for Training routine

    Args:
        :n_epochs: (int): number of epochs.
        batch_size_params (int): number of different parameters instances populating a batch.
        batch_size_timesteps (int): length of the subtrajectories populating a batch.
        batch_size_nodes (int): number of sampled spatial nodes instances populating a batch, needed for NLDFs.
        optimizer (str): type of optimizer (Adam, Adadelta, Adamax, RMSprop, SGD).
        lr (float): learning rate.
        lr_scheduler (tuple[str,float]): type of lr scheduler and decay factor ('ExponentialLR', 0.999)
        lambda_l2 (float): L2 regularization factor.
        warmup_n_epochs (int): number of warmup epochs.
        warmup_lr (float): warmup learning rate.
        validation (bool): flag to enable validation during training.
        bidirectional (bool): flag to enable bidirectional training algorithm.
        enable_tf32 (bool): flag to enable tensorfloat32 precision.
        enable_amp (bool): flag to enable automatic mixed precision.
        compile_mode (bool): flag to enable torch compiling.
        output_directory (str): directory path for storing output metrics.
        metrics_filename (str): metrics filename.
        save_best_checkpoints (bool): flag to enable model checkpoints saves.
        save_checkpoints_freq (int): frequency at which model checkpoints are saved.
    '''
    n_epochs: int or list[int]
    batch_size_params: int
    batch_size_timesteps: int
    batch_size_nodes: int = None
    optimizer: str = 'Adam'
    lr: float or list[float] = 1e-3
    lr_scheduler: tuple[str,float] = None
    lambda_l2: float = 0.    
    warmup_n_epochs: int = None
    warmup_lr: float = 1e-5
    validation: bool = False
    latent_guidance: bool = True
    bidirectional: bool = False
    enable_tf32: bool = False
    enable_amp: bool = False
    compile_mode: str = None
    output_directory: str = None
    metrics_filename: str = None
    save_best_checkpoints: bool = False
    save_checkpoints_freq: int = None
    device: torch.device = torch.device('cpu')

    



[docs]class Trainer:
    '''
    Trainer class, intended to be used with NLDMBase derived classes.

    Args:
        :model: (NLDMBase, NLDM, NLDF), NLDMBase-derived model to be trained.
        :dataset: (DataLoader), training dataset loader.
        :config: (TrainerConfig), trainer configuration.
    '''
    def __init__(self, model: NLDM or NLDF, dataset: DataLoader, config: TrainerConfig):
        
        if isinstance(model,NLDF) and config.batch_size_nodes is None:
            raise ValueError('When training an NLDF batch_size_nodes (int) must be specified via TrainerConfig.')

        if isinstance(model,NLDM) and config.batch_size_nodes is not None:
            warnings.warn('Since the model is of NLDM type, not NLDF, batch_size_nodes is ignored.')
                
        if not issubclass(type(model), NLDMBase):
            raise ValueError(f'model type must be a subclass of NLDMBase (e.g. NLDM or NLDF), got instead {type(model)}.')
        
        self.model = model
        self.dataset = dataset
        self.__parse_config(config)

        self.__set_output_directory()
        if self.output_directory is not None:
            config.save(self.output_directory+'/train_config.json')


    def __parse_config(self,config: TrainerConfig) -> None:

        if isinstance(config.n_epochs, list) and isinstance(config.lr, list):
            assert len(config.n_epochs)==len(config.lr), f'n_epochs and lr lists must have the same length in the provided TrainerConfig, found {len(config.n_epochs),len(config.lr)}'
            assert all(isinstance(x, int) for x in config.n_epochs), f'n_epochs list contains non-integer values.'
        elif not isinstance(config.n_epochs, int) or not isinstance(config.lr, float):
            raise TypeError('n_epochs and lr must be both (int,float) or (list(int),list(float)).')
        
        if type(config.optimizer) in _optim_dict.values():
            self.optimizer = config.optimizer
        elif config.optimizer in _optim_dict.keys():
            self.optimizer_id = config.optimizer
        else:
            raise ValueError(f'Requested optimizer not available. Must be among {list(_optim_dict.keys())}.')
        
        if config.lr_scheduler is not None and config.lr_scheduler[0] not in _lr_scheduler_dict.keys():
            raise ValueError(f'Requested learning rate scheduler not available. Must be among {list(_lr_scheduler_dict.keys())}.')

        if config.bidirectional and self.model.nde_type == 'sde':
            raise ValueError('Bidirectional training mode is not supported for stochastic NLDMs.')
        
        if config.enable_tf32 and config.enable_amp:
            raise ValueError('Conflict between selected precisions, only one flag between "enable_tf32" and "enable_amp" can be enabled.')

        self.n_epochs = config.n_epochs
        self.batch_sizes = [config.batch_size_params, config.batch_size_timesteps, config.batch_size_nodes]
        self.lr = config.lr
        self.lr_scheduler = config.lr_scheduler
        self.lambda_l2 = config.lambda_l2
        self.warmup_lr = config.warmup_lr
        self.warmup_n_epochs = config.warmup_n_epochs
        self.validation = config.validation
        self.latent_guidance = config.latent_guidance
        self.bidirectional = config.bidirectional
        self.enable_tf32 = config.enable_tf32
        self.enable_amp = config.enable_amp
        self.compile_mode = config.compile_mode
        self.output_directory = config.output_directory
        self.metrics_filename = config.metrics_filename
        self.save_best_checkpoints = config.save_best_checkpoints
        self.save_checkpoints_freq = config.save_checkpoints_freq
        self.device = config.device

    def __set_output_directory(self) -> None:
        if self.output_directory is not None:
            if not os.path.isdir(self.output_directory):
                os.makedirs(self.output_directory)

    def __set_device(self) -> None:
        self.dataset.to(self.device)
        self.model.to(self.device)

    def __set_metrics_logger(self) -> None:
        if self.validation:
            self.metrics_logger = MetricsLogger(['epoch','train_loss','val_loss','train_epoch_time','val_epoch_time', 'lr'])
            self.checkpoints_metric_key = 'val_loss'
        else:
            self.metrics_logger = MetricsLogger(['epoch','train_loss','train_epoch_time','lr'])
            self.checkpoints_metric_key = 'train_loss'

    def __set_tf32_backend(self) -> None:
        torch.backends.cuda.matmul.allow_tf32 = self.enable_tf32
        torch.backends.cudnn.allow_tf32 = self.enable_tf32
    
    def __set_model_internals(self) -> None:
        if self.model.training:
            self.model.eval()
        self.model.latent_guidance = self.latent_guidance
        self.model.bidirectional = self.bidirectional
        self.model.train()

    def __run_training_epoch(self, training_epoch) -> None:
        train_epoch_loss, train_epoch_time = timing(training_epoch)(self.dataset, self.optimizer, self.batch_sizes, self.enable_amp)
        self.metrics_logger.update('train_loss', train_epoch_loss)
        self.metrics_logger.update('train_epoch_time', train_epoch_time)
        self.metrics_logger.update('lr', self.optimizer.param_groups[0]['lr'])

    def __run_validation_epoch(self, validation_epoch) -> None:
        if self.validation:
            val_epoch_loss, val_epoch_time = timing(validation_epoch)(self.dataset, self.batch_sizes, self.enable_amp)
            self.metrics_logger.update('val_loss', val_epoch_loss)
            self.metrics_logger.update('val_epoch_time', val_epoch_time)
        
    def __save_model_checkpoints(self,filename,epoch) -> None:
        if self.save_best_checkpoints and self.output_directory is not None:
            if hasattr(self.model,'scaler'):
                torch.save({
                    'epoch': epoch,
                    'lr': self.optimizer.param_groups[0]['lr'],
                    'optimizer': self.optimizer,
                    'model': self.model.state_dict(),
                    'nde_integration_method': self.model.nde_integration_method,
                    'scaler_type': self.model.scaler.type,
                    'scaler_params': self.model.scaler.params
                }, os.path.join(self.output_directory,filename))
            else:
                torch.save({
                    'epoch': epoch,
                    'lr': self.optimizer.param_groups[0]['lr'],
                    'optimizer': self.optimizer,
                    'model': self.model.state_dict(),
                    'nde_integration_method': self.model.nde_integration_method
                }, os.path.join(self.output_directory,filename))


[docs]    def train(self, verbose=True) -> None:
        '''
        Method starting the training routine.
        '''
        self.__set_metrics_logger()
        self.__set_device()
        self.__set_tf32_backend()        
        self.__set_model_internals()
        
        training_epoch = torch.compile(self.model.training_epoch, mode=self.compile_mode) if self.compile_mode is not None else self.model.training_epoch
        validation_epoch = torch.compile(self.model.validation_epoch, mode=self.compile_mode) if self.compile_mode is not None else self.model.validation_epoch

        best_loss = math.inf
        best_epoch = 0
        epoch_index = 0  # global epochs index
        tot_start_time = time.time()

        if None not in (self.warmup_lr, self.warmup_n_epochs):

            if verbose:
                print('Training Warm Up\n')
            
            try:
                # check if optimizer passed to the configuration else define a new one
                self.optimizer
            except AttributeError:
                self.optimizer = _optim_dict[self.optimizer_id](self.model.parameters(),lr=self.warmup_lr, weight_decay=self.lambda_l2)
                    
            for _ in range(self.warmup_n_epochs):
                
                self.metrics_logger.update('epoch', epoch_index)
                self.__run_training_epoch(training_epoch)

                if verbose:
                    self.metrics_logger.print_current()
                
                epoch_index += 1
        
        if verbose:
            print('\nTraining\n')

        n_epochs_list = self.n_epochs
        lr_list = self.lr
        
        if isinstance(n_epochs_list, int) and isinstance(lr_list, float):
            n_epochs_list = [n_epochs_list]
            lr_list = [lr_list]
        
        try:
            self.optimizer
        except AttributeError:
            # define optimizer if training warmup was not performed or not passed to the configuration
            self.optimizer = _optim_dict[self.optimizer_id](self.model.parameters(), lr=lr_list[0], weight_decay=self.lambda_l2)

        if self.lr_scheduler is not None:
            scheduler = _lr_scheduler_dict[self.lr_scheduler[0]](optimizer=self.optimizer, gamma=self.lr_scheduler[1])

        for (n_epochs,lr) in zip(n_epochs_list,lr_list):
            
            self.optimizer.param_groups[0]['lr'] = lr

            for _ in range(n_epochs):
                
                self.metrics_logger.update('epoch', epoch_index)

                self.__run_training_epoch(training_epoch)
                self.__run_validation_epoch(validation_epoch)
                
                if self.save_best_checkpoints: 
                    # save best checkpoints
                    if self.metrics_logger.current_metrics[self.checkpoints_metric_key] < best_loss:
                        best_loss = self.metrics_logger.current_metrics[self.checkpoints_metric_key]
                        best_epoch = epoch_index
                        self.__save_model_checkpoints('checkpoint.pt',epoch_index)

                if self.save_checkpoints_freq is not None and epoch_index%self.save_checkpoints_freq==0:
                    # save checkpoints given saving frequency
                    self.__save_model_checkpoints(f'checkpoint_{epoch_index}.pt',epoch_index)

                if self.metrics_filename is not None:
                    # save metrics file
                    self.metrics_logger.save(os.path.join(self.output_directory,self.metrics_filename))

                if verbose:
                    # print metrics
                    self.metrics_logger.print_current()

                if self.lr_scheduler is not None:
                    # update learning rate if scheduler is specified
                    scheduler.step()

                epoch_index += 1

            if verbose:
                print('\n')
                
        if verbose:
            tot_training_time = round(time.time()-tot_start_time,5)
            print(f'Best validation epoch: {best_epoch}, validation loss: {best_loss}, total training time: {tot_training_time} s')
