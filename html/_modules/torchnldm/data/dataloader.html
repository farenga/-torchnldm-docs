[docs]class DataLoader:
    '''
    DataLoader class. It allows to load and manage PDEGen (https://github.com/farenga/pdegen/) datasets, or .hdf5 files complying
    with PDEGen dataset structure. 
        
    Args:
        :config: (DataLoaderConfig), dataset configuration object.
    '''
    def __init__(self, config: DataLoaderConfig, verbose=True) -> None:
        
        # set dataset device (cpu/gpu) and path
        self.device = config.device
        self.filepath = config.filepath

        # load tensors in hdf5 file
        self.__load_tensors()

        if config.time_subsampling is not None:
            # parsing subsampling configuration (Nt, subsampling_method)
            self.__time_subsampling(config.time_subsampling[0],config.time_subsampling[1])

        if config.time_validation_split is not None:
            # train/test splitting along time dimension
            if isinstance(config.time_validation_split, int):
                assert config.time_validation_split < self.Nt, f'When providing an int time_validation_split train/test time-split index must be <Nt, got instead {config.time_validation_split}'
                self.Nt_train = config.time_validation_split
            elif isinstance(config.time_validation_split, float):
                assert config.time_validation_split < 1., f'When providing a float time_validation_split, it must be <1 since, got instead {config.time_validation_split}'
                self.Nt_train = int(config.time_validation_split*self.Nt)
            else:
                raise TypeError(f'time_validation_split type must be either int (<Nt) or float (<1.), got instead {type(config.time_validation_split)}.')
            
            self.sections.extend(['train','validation'])

        if config.reduced_dim is not None:
            if isinstance(config.reduced_dim, int):
                self.reduced_dim = config.reduced_dim
                self.__reduce()
            else:
                raise TypeError(f'reduced_dim type must be int, got insteast {type(config.reduced_dim)}')


        if config.scaler_type is not None:
            # rescale data based on specified method
            # scaler params are orderd in the following manner
            # minmax:   [S_min, S_max, P_min, P_max, coords_min, coords_max]
            # standard: [S_mean, S_std, P_mean, P_std, coords_mean, coords_std]
            self.scaler_type = config.scaler_type
            self.__scale()

        if verbose:    
            print('Dataset loading complete.')
            self.summary()
        
    def __load_tensors(self) -> None:
        h5file = h5py.File(self.filepath, 'r')
        self.__S = torch.from_numpy(h5file['S'][:]).to(self.device).contiguous() 
        self.__P = torch.from_numpy(h5file['P'][:]).to(self.device).contiguous() 
        self.__C = torch.from_numpy(h5file['C'][:]).to(self.device).contiguous()
        self.sections = ['complete']

        N, Nt, _ = self.__S.shape
        N_, Nt_, _ = self.__P.shape       

        assert N==N_, f'First dimension (N) of S and P tensors must match, got instead: {N}!={N_}'
        assert Nt==Nt_, f'Second dimension (Nt) of S and P tensors must match, got instead: {Nt}!={Nt_}'

        dims = h5file['dims'].attrs         
        self.N, self.Nt, self.Nh, self.Np, self.n = dims['N'], dims['Nt'], dims['Nh'], dims['Np'], dims['n']
        self.M, self.d = dims['M'], dims['d']

    def __time_subsampling(self,  Nt_new:int, subsampling_method: str='regular') -> None:
        assert Nt_new < self.Nt, f'Sub-sampling Nt_new must be < Nt={self.Nt}, got instead {Nt_new}.'
        if subsampling_method=='regular':
            subsampled_indices = list(range(0,self.Nt,self.Nt//Nt_new))
        elif subsampling_method=='random':
            subsampled_indices = np.sort(np.random.choice(np.arange(self.Nt), Nt_new, replace=False))
        else:
            raise ValueError(f'Admissible subsampling types are either "regular" or "random", got instead {subsampling_method}.')
        
        self.__S = self.__S[:,subsampled_indices,:]
        self.__P = self.__P[:,subsampled_indices,:]
        self.Nt = Nt_new
    
    def __reduce(self) -> None:
        '''
        Substitutes the snapshots tensor S with the projected one via POD.
        '''
        self.__S, self.U = POD(self.__S, self.reduced_dim)

    def __scale(self) -> None:
        '''
        Instantiates a DataScaler object and performs scaling of the dataset's features.
        '''
        from .datascaler import DataScaler
        self.scaler = DataScaler(self.scaler_type, self)
        self.scaler.scale_dataset(self)

        '''
        Prints a summary of the dataset's infos.
        '''
        print('\nDataset Summary')
        print(' Dataset location:', self.filepath)
        print(' Dimensions:\tN:', self.N,', Nt:', self.Nt,', Np:', self.Np,', Nh:', self.Nh,', n:', self.n )
        print('\t\tM:', self.M, ', d:', self.d)

        if hasattr(self,'Nt_train'):
            print(' Train-validation time splitting (Nt_train):', self.Nt_train)
        if hasattr(self,'scaler_type'):
            print(' Scaler type:', self.scaler_type)
        if hasattr(self,'reduced_dim'):
            print(' POD modes:', self.reduced_dim)

[docs]    def get_section(self, section_id:str='complete') -> tuple[Tensor,Tensor,int]:
        '''
        Returns a section of the dataset.
        Instead of creating multiple tensors, accounting for the train and validation splits, the splits
        are managed via indexing over the time-dimension.
        Args:
            :section_id: (str): name of the dataset subsection ('complete','train','validation').
        Returns:
            S (Tensor), P (Tensor), max_timesteps (int).
        '''        
        assert section_id in self.sections, f'section_id must be present in {self.sections}, got instead {section_id}.'
        
        if section_id=='complete':
            return self.__S, self.__P, self.Nt
        elif section_id=='train':
            return self.__S[:,:self.Nt_train,:], self.__P[:,:self.Nt_train,:], self.Nt_train
        elif section_id=='validation':
            return self.__S[:,self.Nt_train-1:,:], self.__P[:,self.Nt_train-1:,:], self.Nt - self.Nt_train

        

[docs]    def get_train_t0_indices(self, timesteps: int, shuffle: bool = False) -> tuple[int,list[int],int]:
        '''
        Args:
            timesteps (int): number of timesteps (sequence length)
        Returns:
            section_id
        '''     
        if 'train' in self.sections:
            section_id = 'train'    
            n_subtrajectories = self.Nt_train - timesteps
        else:
            section_id = 'complete'
            n_subtrajectories = self.Nt - timesteps
        
        sub_trajectories_t0_indices = list(range(n_subtrajectories))
        
        if shuffle:
            random.shuffle(sub_trajectories_t0_indices)

        return section_id, sub_trajectories_t0_indices, n_subtrajectories 



[docs]    def get_trajectory(self,
                section_id: str,
                t0_index: int,
                param_index: int,
                timesteps: int,
                field_form: bool = False) -> tuple[Tensor]:


        '''
        Returns a single trajectory (ut), their initial values (u0) the respective parameters 
        instances (mut0_batch), and timesteps (t_batch), sampled from the Snapshots and Parameters 
        tensors (S,P), given batch sizes an initial timestep. In the case batch_size_nodes is passed, 
        the output will be in field form, so the last dimension of the state trajectory tensor will be 
        of shape (batch_size_nodes,n) instead than Nh.
        
        Args:
            :section_id: (str), id of the dataset section ('complete','train','validation').
            :t0_index: (int), index of the initial timestep (<Nt-batch_size_timesteps)
            :param_index: (int), index of sampled paramter instance (<=N)
            :timesteps: (int), trajectories length (<=Nt)
            :field_form: (bool), flag enabling field form

        Returns:
            :u0: (Tensor), initial value, shape = (Nh,) or (M,n)
            :mut0: (Tensor), parameters instances and timesteps related to the initial value, shape = (Np)
            :x: (Tensor), mesh nodes coordinates, shape = (M, d)
            :t: (Tensor), 1d tensor of timesteps, shape = (batch_size_timesteps,)
            :ut: (Tensor), state trajectory, shape = (timesteps, Nh) or (...,batch_size_nodes,n)

        '''

        # Checks
        assert section_id in self.sections, f'Data selector must be present in {self.sections}'
        assert param_index < self.N, 'param_index must be <N.'
        
        # Extracting dataset subsection        
        S,P, max_timesteps = self.get_section(section_id)
        
        assert t0_index < max_timesteps - timesteps, f'Must be: t0_index < max_timesteps - timesteps, got instead {t0_index} >= {max_timesteps} - {timesteps}'
        
        # Extracting trajectory, initial value, parameters
        ut = S[param_index,t0_index:t0_index+timesteps,:] # [timesteps,Nh]
        u0 = ut[0,...]                                    # [Nh]
        mut0 = P[param_index,t0_index,:]                  # [Np+1]
        t = P[0,t0_index:t0_index+timesteps,-1]           # [timesteps]
        x = self.__C                                        # [M,d]
        
        if field_form:
            # Reshape last dimension for Field formulation
            # such that (...,Nh)->(...,M,n)
            ut = ut.view(timesteps, self.M, self.n)        # [timesteps, M, n]
            u0 = u0.view(self.M, self.n)                   # [M, n]
            #mut0 = mut0.unsqueeze(0).repeat(self.M, 1)     # [M, Np+1]
            mut0 = mut0.unsqueeze(0).expand(self.M, -1)     # [M, Np+1]

        return  u0.contiguous(), mut0.contiguous(), x.contiguous(), t.contiguous(), ut.contiguous()



[docs]    def get_batch(self,
                  section_id: str,
                  t0_index: int,
                  batch_size_params: int,
                  batch_size_timesteps: int, 
                  batch_size_nodes: int = None) -> tuple[Tensor]:

        '''
        Returns a batch of trajectories (ut), their initial values (u0) the respective parameters 
        instances (mut0_batch), and timesteps (t_batch), sampled from the Snapshots and Parameters 
        tensors (S,P), given batch sizes an initial timestep. In the case batch_size_nodes is passed, 
        the output will be in field form, so the last dimension of the state trajectory tensor will be 
        of shape (batch_size_nodes,n) instead than Nh.
        
        Args:
            :section_id: (str), id of the dataset section ('complete','train','validation').
            :t0_index: (int), index of the initial timestep (<Nt-batch_size_timesteps)
            :batch_size_params: (int), number of sampled trajectories (<=N)
            :batch_size_timesteps: (int), trajectories length (<=Nt)
            :batch_size_nodes: (int), number of sampled mesh nodes (<=M)

        Returns:
            :u0: (Tensor), batch of initial values, shape = (batch_size_params, Nh) or (...,batch_size_nodes,n)
            :mut0: (Tensor), batch of parameters and timesteps related to the initial values, shape = (batch_size_params, Np)
            :x: (Tensor), batch of sampled mesh nodes coordinates, shape = (batch_size, batch_size_nodes, d)
            :t: (Tensor), 1d tensor of timesteps, shape = (batch_size_timesteps,)
            :ut: (Tensor), batch of state trajectories, shape = (batch_size_timesteps, batch_size_params, Nh) or (...,batch_size_nodes,n)
            :mut: (Tensor), batch of parameters and timesteps related to the whole trajectory, shape = (batch_size_timesteps, batch_size, Np)

        '''

        # Checks
        assert section_id in self.sections, f'Data selector must be present in {self.sections}'
        assert 1<= batch_size_params <= self.N, 'batch_n_param must be >=1, <=N.'

        # Extracting dataset subsection        
        S,P, max_timesteps = self.get_section(section_id)
        assert 1<= batch_size_timesteps <= self.Nt, 'batch_n_param must be >=1, <=Nt.'

        assert t0_index < max_timesteps - batch_size_timesteps, f'Must be: t0_index < max_timesteps - batch_size_timesteps, got instead {t0_index} >= {max_timesteps} - {batch_size_timesteps}'
        
        params_sampled_indices = np.array([0]) if self.N==1 else np.random.choice(np.arange(self.N), batch_size_params, replace=False)
            
        # Forward trajectories extraction 
        # permuting to have time as first axis in batch tensors
        # (torch.permute gives a speedup wrt torch.stack, since we avoid the creation of an array of tensors)
        tf_index = t0_index+batch_size_timesteps
        ut = torch.permute(S[params_sampled_indices if batch_size_params>1 else ...,t0_index:tf_index,:],(1,0,2))   # [batch_size_timesteps, batch_size_params, Nh]
        mut = torch.permute(P[params_sampled_indices if batch_size_params>1 else ...,t0_index:tf_index,:],(1,0,2))  # [batch_size_timesteps, batch_size_params, Np]
        t = P[0,t0_index:tf_index,-1]                                                                               # [batch_size_timesteps]
        x = self.__C                                                                                                # [M, d]

        if batch_size_nodes is not None:
            # Sampling grid nodes coordinates and reshaping tensors for Field formulation
            assert batch_size_nodes <= self.M, 'batch_size_nodes must be <=M.'
            nodes_sampled_indices = np.random.choice(np.arange(self.M-batch_size_nodes), 1) + np.arange(batch_size_nodes)
            ut = ut.view(batch_size_timesteps, batch_size_params, self.M, self.n)[:,:, nodes_sampled_indices, :] # [batch_size_timesteps, batch_size_params, batch_size_nodes, n]
            #mut = mut.unsqueeze(-2).repeat(1, 1, batch_size_nodes, 1)                                           # [batch_size_timesteps, batch_size_params , batch_size_nodes, Np]
            #x = self.__C[nodes_sampled_indices, :].unsqueeze(0).repeat(batch_size_params, 1, 1)                 # [batch_size_params, batch_size_nodes, d]
            mut = mut.unsqueeze(-2).expand(-1, -1, batch_size_nodes, -1)                                         # [batch_size_timesteps, batch_size_params , batch_size_nodes, Np]
            x = self.__C[nodes_sampled_indices, :].unsqueeze(0).expand(batch_size_params, -1, -1)                # [batch_size_params, batch_size_nodes, d]
            
        u0 = ut[0,...]                                                                                 # [batch_size_params, Nh] or [batch_size_params, batch_size_nodes, n]
        mut0 = mut[0,...]                                                                              # [batch_size_params, Np] or [batch_size_params, batch_size_nodes, Np]

        return  u0.contiguous(), mut0.contiguous(), x.contiguous(), t.contiguous(), ut.contiguous(), mut.contiguous()
